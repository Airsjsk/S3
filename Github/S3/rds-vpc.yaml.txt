AWSTemplateFormatVersion: "2010-09-09"
Description: VPC with three subnets and an RDS instance

Parameters:
  StudentID:
    Type: String
    Description: Student ID used for naming resources

Resources:
  MyVPC:
    Type: AWS::EC2::VPC
    Properties:
      CidrBlock: 10.0.0.0/16
      EnableDnsSupport: true
      EnableDnsHostnames: true
      Tags:
        - Key: Name
          Value: !Sub "VPC-${StudentID}"

  SubnetA:
    Type: AWS::EC2::Subnet
    Properties:
      VpcId: !Ref MyVPC
      CidrBlock: 10.0.1.0/24
      AvailabilityZone: us-west-2a
      MapPublicIpOnLaunch: true
      Tags:
        - Key: Name
          Value: !Sub "SubnetA-${StudentID}"

  SubnetB:
    Type: AWS::EC2::Subnet
    Properties:
      VpcId: !Ref MyVPC
      CidrBlock: 10.0.2.0/24
      AvailabilityZone: us-west-2b
      MapPublicIpOnLaunch: true
      Tags:
        - Key: Name
          Value: !Sub "SubnetB-${StudentID}"

  SubnetC:
    Type: AWS::EC2::Subnet
    Properties:
      VpcId: !Ref MyVPC
      CidrBlock: 10.0.3.0/24
      AvailabilityZone: us-west-2c
      MapPublicIpOnLaunch: true
      Tags:
        - Key: Name
          Value: !Sub "SubnetC-${StudentID}"

  RdsSecurityGroup:
    Type: AWS::EC2::SecurityGroup
    Properties:
      GroupDescription: Security group for RDS instance
      VpcId: !Ref MyVPC

  RdsSubnetGroup:
    Type: AWS::RDS::DBSubnetGroup
    Properties:
      DBSubnetGroupDescription: Subnet group for RDS instance
      SubnetIds:
        - !Ref SubnetA
        - !Ref SubnetB
        - !Ref SubnetC
      DBSubnetGroupName: !Sub "rds-subnet-group-${StudentID}"

  RdsInstance:
    Type: AWS::RDS::DBInstance
    Properties:
      AllocatedStorage: "5"
      DBInstanceClass: db.t3.micro
      DBInstanceIdentifier: !Sub "playground-db-${StudentID}"
      Engine: MySQL
      MasterUsername: Admin
      MasterUserPassword: CloudAcademy123!
      VPCSecurityGroups:
        - !GetAtt RdsSecurityGroup.GroupId
      DBSubnetGroupName: !Ref RdsSubnetGroup
      PubliclyAccessible: false
      BackupRetentionPeriod: 0

Outputs:
  VpcId:
    Description: VPC ID
    Value: !Ref MyVPC

  SubnetIds:
    Description: Comma-delimited list of subnet IDs
    Value: !Join
      - ","
      - - !Ref SubnetA
        - !Ref SubnetB
        - !Ref SubnetC
